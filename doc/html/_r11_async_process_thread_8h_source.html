00105 inline void R11AsyncProcessThread::Sync()
00106 {
00107   _resumeMutex.lock();
00108 
00109   if( !_gotSync ) // if haven't already got sync

00152 inline void R11AsyncProcessThread::_ThreadTick()
00153 {
00154   while( !_stop )
00155   {
00156     _resumeMutex.lock();
00157 
00158     // notify sync
00159     _gotSync = true; // set the sync flag
00160     _syncCondt.notify_all();
00161 
00162     // wait for resume
00163     if( !_gotResume ) // if haven't already got resume
00164     {
00165       _resumeCondt.wait( _resumeMutex ); // wait for resume
00166     }
00167     _gotResume = false; // reset the resume flag
00168 
00169     _resumeMutex.unlock();
00170 
00171     if( !_stop && _tickMethod != nullptr )
00172     {
00173       _tickMethod( 0 );
00174     }
00175   }
00176 
00177   // set _stopped flag (notify _Stop that the thread has exited)
